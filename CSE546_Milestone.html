# Model Fit Solution 1: Use Multiple linear regression

# Purpose of this model is to get a more intuitive sense of the impact of the predictors. 
# Right-censored cases are partially ignored. Items that are listed for more than 15 days 
# (the average days to sell) are included in the model. Items that are listed fewer than 15 days are 
# not included because we don't want to underestimate the days. 
# This model is inherently biased. The survival model (Cox Regression) will deal with this better. 

features_p = [ 'brand_tier_id', 'category_tier_id',
        'new_with_tags',
       'quality_code', 'merchandising_department',
        'payout_policy', 'relisted',
       'days_box','box_discount', 'box_original', 'box_msrp']

data2 = data_sold[features_p]

    real = validate['days']
    error = sum(abs(predictions - real))/len(real)
    Error.append(error)

    RSS = sum((predictions- real)**2)
    TSS = sum((real - np.mean(real))**2)
    r= 1-RSS/TSS
    R.append(r)
